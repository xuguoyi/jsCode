# 第二章，表单常用代码
## 1，去除字符串左右两边的空格
```html
	<input type="text" id="strs" value="     需要过滤空格  " />
	<input type="button" id="rstrsBtn" value="过滤" />
```
```js
	var $ = function(el){
		return document.getElementById(el)
	}
	$("rstrsBtn").onclick = function(){
		$("strs").value = $("strs").value.replace(/^(\s|\u00A0)+|(\s|\u00A0)+$/g,'')
	}
```
